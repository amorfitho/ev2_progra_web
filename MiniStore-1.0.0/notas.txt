const addToCartButton = document.querySelector('#add-to-cart-button');
addToCartButton.addEventListener('click', () => {
  // Add the product to the shopping cart
  addProductToCart();
});

function addProductToCart() {
  // Get the product details
  const productName = 'Product Name';
  const productPrice = 19.99;
  const productImage = 'https://example.com/product-image.jpg';

  // Send the product details to the shopping cart
  sendProductToCart(productName, productPrice, productImage);
}

function sendProductToCart(productName, productPrice, productImage) {
  // Send the product details to the shopping cart using AJAX or fetch API
  // For example:
  fetch('/add-to-cart', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      productName,
      productPrice,
      productImage
    })
  })
  .then(response => response.json())
  .then(data => {
    // Update the shopping cart UI
    updateCartUI(data);
  })
  .catch(error => {
    console.error('Error:', error);
  });
}

function updateCartUI(data) {
  // Update the shopping cart UI with the new product details
  // For example:
  const cartList = document.querySelector('#cart-list');
  const newProduct = document.createElement('li');
  newProduct.innerHTML = `
    <img src="${data.productImage}" alt="${data.productName}">
    <div>
      <h3>${data.productName}</h3>
      <p>${data.productPrice}</p>
    </div>
  `;
  cartList.appendChild(newProduct);
}
